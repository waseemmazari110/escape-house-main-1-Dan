### Touch Gesture Detection:
```tsx
// Minimum swipe distance
const minSwipeDistance = 50;

// Detect left swipe (next)
if (touchStart - touchEnd > minSwipeDistance) {
  goToNext();
}

// Detect right swipe (previous)
if (touchStart - touchEnd < -minSwipeDistance) {
  goToPrevious();
}
```
